public VoxelShape makeShape(){
	VoxelShape shape = Shapes.empty();
	shape = Shapes.join(shape, Shapes.box(0.25, 0.25, 0.25, 0.9375, 0.75, 0.75), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0.9375, 0.1875, 0.1875, 1, 0.8125, 0.8125), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0.1875, 0.1875, 0, 0.8125, 0.8125, 0.0625), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0.25, 0.25, 0.0625, 0.75, 0.75, 0.25), BooleanOp.OR);

	return shape;
}
